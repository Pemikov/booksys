    <script>
        let currentDate = new Date();
        let availableSlots = [];
        let selectedSlot = null;
        let services = [];

        function formatDate(date) { return date.toISOString().split('T')[0]; }
        function formatDisplayDate(date) { return date.toLocaleDateString('en-US', { weekday: 'long', year: 'numeric', month: 'long', day: 'numeric' }); }
        
        function changeDate(delta) { currentDate.setDate(currentDate.getDate() + delta); loadCalendar(); }
        
        async function loadCalendar() {
            document.getElementById('currentDate').textContent = formatDisplayDate(currentDate);
            document.getElementById('calendarGrid').innerHTML = '<div class="loading">Loading calendar...</div>';
            const date = formatDate(currentDate);
            try {
                const response = await fetch(`/api/calendar/availability/${date}?orgId=1`);
                const data = await response.json();
                availableSlots = data.slots || [];
                renderCalendar();
            } catch (error) {
                document.getElementById('calendarGrid').innerHTML = '<div class="loading">Error loading calendar. Please try again.</div>';
            }
        }
        
        function renderCalendar() {
            if (!availableSlots.length) {
                document.getElementById('calendarGrid').innerHTML = '<div class="loading">No available slots for this day</div>';
                return;
            }
            const html = `<table class="daily-view"><thead><tr><th>Time</th><th>Status</th><th>Staff</th><th>Action</th></tr></thead><tbody>
                ${availableSlots.map(slot => `<tr class="time-slot ${slot.is_available ? 'available' : 'booked'}">

                    <td>${slot.is_available ? '✅ Available' : '❌ Booked'}</td>
                    <td>${slot.staff_name || 'Any'}</td>
                    <td>${slot.is_available ? `<button onclick="openBookingModal('${slot.slot_start}', '${slot.slot_end}')">Book</button>` : 'Not available'}</td>
                </tr>`).join('')}
            </tbody></table>`;
            document.getElementById('calendarGrid').innerHTML = html;
        }
        
        async function loadServicesAndStaff() {
            try {
                const [servicesRes, staffRes] = await Promise.all([
                    fetch('/api/calendar/services?orgId=1'),
                    fetch('/api/calendar/staff?orgId=1')
                ]);
                const servicesData = await servicesRes.json();
                const staffData = await staffRes.json();
                services = servicesData.services || [];
                staff = staffData.staff || [];
                document.getElementById('serviceSelect').innerHTML = services.map(s => `<option value="${s.id}">${s.name} (${s.duration_minutes} min - $${s.price})</option>`).join('');
                document.getElementById('staffSelect').innerHTML = '<option value="">Any staff</option>' + staff.map(s => `<option value="${s.id}">${s.name} - ${s.role}</option>`).join('');
            } catch (error) { console.error('Error loading data:', error); }
        }
        
        function openBookingModal(startTime, endTime) {
            selectedSlot = { startTime, endTime };
            document.getElementById('selectedSlot').innerHTML = `<strong>Time:</strong> ${startTime} - ${endTime}<br><strong>Date:</strong> ${formatDisplayDate(currentDate)}`;
            document.getElementById('bookingModal').style.display = 'flex';
            document.getElementById('modalMessage').innerHTML = '';
        }
        
        async function confirmBooking() {
            const customerName = document.getElementById('customerName').value;
            const customerEmail = document.getElementById('customerEmail').value;
            const customerPhone = document.getElementById('customerPhone').value;
            const serviceId = document.getElementById('serviceSelect').value;
            const staffId = document.getElementById('staffSelect').value;
            if (!customerName || !customerEmail) {
                document.getElementById('modalMessage').innerHTML = '<div class="error">Please fill in name and email</div>';
                return;
            }
            const booking = {
                organization_id: 1,
                customer_name: customerName,
                customer_email: customerEmail,
                customer_phone: customerPhone,
                booking_date: formatDate(currentDate),
                start_time: selectedSlot.startTime,
                end_time: selectedSlot.endTime,
                service_id: serviceId || null,
                staff_id: staffId || null
